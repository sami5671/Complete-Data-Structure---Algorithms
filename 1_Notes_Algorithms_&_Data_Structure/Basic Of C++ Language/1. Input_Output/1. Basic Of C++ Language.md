### Add Two Numbers (User Input)

```C++
#include<iostream>

using namespace std;

int main(){

    int a,b, sum;

    cout<< "Enter number 1: ";
    cin >> a;

    cout<< "Enter number 2: ";
    cin>> b;

    sum = a + b;

    cout<<"The of "<< a << " & " << b << " is: " << sum << endl;

    return 0;

}
```
